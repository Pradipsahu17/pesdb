CREATE TABLE IF NOT EXISTS atm.accepted_bank(
	accepted_bank_id serial PRIMARY KEY UNIQUE NOT NULL,
	bin_id character varying UNIQUE NOT NULL,
	bank_name character varying NOT NULL,
	charge money NOT NULL,
	bal_inquiry_charge money NOT NULL,
	coop_withdrawl_charge money DEFAULT '0' NOT NULL,
	coop_bal_inquiry_charge money DEFAULT '0' NOT NULL,
	status boolean NOT NULL DEFAULT true,
	audit_user_id integer REFERENCES office.users(user_id) NOT NULL,
	audit_ts timestamp with time zone DEFAULT now()
);
